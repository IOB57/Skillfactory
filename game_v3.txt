import random

def binary_search_guess():
    # Генерируем случайное число
    predict_number = random.randint(1, 1000)
    print(f"Секретное число: {predict_number}")

    # Устанавливаем границы диапазона
    low, high = 1, 1000
    count = 0

    while low <= high:
        count += 1
        # Находим середину диапазона
        guess = (low + high) // 2
        print(f"Попытка {count}: {guess}")

        # Проверяем угадали ли число
        if guess == predict_number:
            print(f"Угадали число {predict_number} за {count} попыток!")
            break
        elif guess < predict_number:
            low = guess + 1  # Число больше текущей догадки
        else:
            high = guess - 1  # Число меньше текущей догадки

    if count > 20:
        print("Не удалось угадать число за 20 попыток")
    return count

# Пример вызова функции
binary_search_guess()
